注意

静态文件在.vuepress/pubulic目录下，配置文件为.vuepress/config.js

# avatar 头像

// .vuepress/config.js
module.exports = {
    themeConfig:{
        avatar: '/avatar.png' 
    }
} ;

# 站点 logo

// .vuepress/config.js
module.exports = {
    themeConfig:{
        logo: '/logo.png' 
    }
} ;

# 站点标题

// .vuepress/config.js
module.exports = {
    title:'hvnobug'
} ;

# 站点描述

// .vuepress/config.js
module.exports = {
    description:'Welcome to my vuepress site!'
} ;
